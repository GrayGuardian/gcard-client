local Template = {}

local getInfos = function(name)
    return require(name)
end

local getMap = function(name, key)
    local map = {}
    local infos = getInfos(name)
    for index, info in ipairs(infos) do
        if info[key] ~= null then
            map[info[key]] = info
        else
            --未找到标识
        end
    end
    return map
end

local template_error_info = getMap('template_error_info', "code");
local template_props = getMap("template_props", 'id');

Template.template_error_info = template_error_info;
Template.template_props = template_props;

return Template
