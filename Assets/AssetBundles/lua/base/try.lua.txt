-- 若success为true，result为try返回值，若success为false，result为错误信息

return function(block)
    local catchFunc = block['catch']
    local finallyFunc = block['finally']

    local arr = {}
    if #block > 0 then
    	for i = 1, #block do
    		table.insert(arr, block[i])
    	end
    end

    local success, result = pcall(unpack(arr))

    if not success and catchFunc then
        catchFunc(result)
    end

    if finallyFunc then
        finallyFunc(success, result)
    end

    if success then
        return result
    end
end
